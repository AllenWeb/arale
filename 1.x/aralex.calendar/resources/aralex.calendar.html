1. 无Select控件日历

示例HTML

<input type="button" value="Show Calendar" id="btn1" />
<input type="text" value="select date from calendar" id="text" />
<div id="cal_cantainer1"></div>


实例代码

$E.domReady(function(){
    var cal = new aralex.calendar.Calendar({
	srcId : 'cal_cantainer1',
	week_start : '7',
	selected : ['2010/10/23','2010/10/29'],
	afterDoSelectEvent : function(date){
		$('text').attr('value',date);
	},
	full_selector:true,
	position : [100,100]
    });
    cal.render();
	
    $('btn1').click(function(e){
	var target = $(e.target);
	if($(e.target).hasClass('show')){
		cal.hide();
		target.removeClass('show');
		target.attr('value','Hide Calendar')
	}else{
		cal.show();
		target.addClass('show');
		target.attr('value','Show Calendar')
	}
    });
});





2. 有Select控件日历

Useful Tips

    

  • 
dropfliter参数必须与mindate,maxdate参数一起使用， 如不设置将默认从2004到当前年区间。

    • 


示例HTML

<input type="button" value="Show Calendar" id="btn1" />
<input type="text" value="select date from calendar" id="text" />
<div id="cal_cantainer"></div>


实例代码

$E.domReady(function(){
    var cal = new aralex.calendar.Calendar({
	srcId : 'cal_cantainer2',
	week_start : '7',
	dropfilter : true,
	mindate  : '1999/11/11',
	maxdate  : '2012/12/11',
	selected : ['2010/10/23','2010/10/29'],
	afterDoSelectEvent : function(date){
		$('text').attr('value',date);
	},
	position : [100,100]
	});
});
