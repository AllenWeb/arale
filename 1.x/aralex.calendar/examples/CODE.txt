<script>
var cal = new aralex.calendar.Calendar({	
	srcId : 'cal_cantainer1',				//页面容器id
	week_start : '7',						//起始星期
	selected : ['2010.10.23'],				//已选择日期
	afterDoSelectEvent : function(date){	//选择日期后的行为，一般用于填充输入框
		$('text').attr('value',date);
	},
	mindate : "2011.3.2",					//最小日期
	maxdate : "2012.3.2",					//最大日期
	date_delimiter : ".",					//日期分隔符
	position : [400,50]						//相对位置
});
cal.render();								//日期对象生成
cal.show();									//显示日期，一般绑定在输入框focus事件上
//cal.hide();
</script>
